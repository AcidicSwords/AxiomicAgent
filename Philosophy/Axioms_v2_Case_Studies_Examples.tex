\section*{Purpose}
Provide \emph{examples} of Axioms v2 in use. Each study applies the same lenses: typed bounds, capacity, turning points, web (edge quality, $\mathrm{TED}_\tau$), handoff contracts, order/tariffs, resistance triage, and a minimal formal stub.

\section{General Relativity: Black Holes}
\textbf{Bounds.} Invariant: causality, conservation, Einstein equations; Enacted: simulation closures; Soft: priors.
\newline
\textbf{Capacity.} Detector SNR, array coherence, FLOPs.
\newline
\textbf{Turning points.} Horizon formation; merger$\to$ringdown; Page-time regimes.
\newline
\textbf{Web.} Causal/entanglement graphs; increase $q$ via calculability and stability; monitor $\mathrm{TED}_\tau$ for inference routes.
\newline
\textbf{Handoffs.} GR $\leftrightarrow$ GRMHD $\leftrightarrow$ QFT; contracts list conserved quantities and admissible approximations.
\newline
\textbf{Order.} Collapse history; ringdown stability (high $s$); reversal near-irreversible.
\newline
\textbf{Triage.} Measurement and model frictions dominate.
\newline
\textbf{Stub.} Commit when $\Theta_{\mathrm{horizon}}(p)\,(p\!\ge\!p^\*)$; ringdown stabilization grows with $s$.
